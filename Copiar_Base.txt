Sub Copiar_Base()

Dim ws As Sheets

A = ThisWorkbook.Sheets("Sheet1").Range("B1").Text
B = ThisWorkbook.Sheets("Sheet1").Range("B2").Text

Application.DisplayAlerts = True
Application.ScreenUpdating = True

End Sub
